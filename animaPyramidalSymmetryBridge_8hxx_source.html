216 void PyramidalSymmetryBridge<PixelType,ScalarType>::ComputeRealignTransform(typename itk::Vector <double,InputImageType::ImageDimension> centralPoint,
217  typename InputImageType::PointType &centerReal, ParametersType &imageParams)
218 {

220  imageMidPlaneTrsf->SetParameters(imageParams);
221  imageMidPlaneTrsf->SetRotationCenter(centerReal);
222 
223  typedef itk::Matrix <ScalarType,InputImageType::ImageDimension+1,InputImageType::ImageDimension+1> HomogeneousMatrixType;
224 
225  HomogeneousMatrixType midPlaneMatrix;
226  midPlaneMatrix.SetIdentity();
227 
228  for (unsigned int i = 0;i < InputImageType::ImageDimension;++i)
229  {
230  midPlaneMatrix(i,InputImageType::ImageDimension) = imageMidPlaneTrsf->GetOffset()[i];
231 
232  for (unsigned int j = 0;j < InputImageType::ImageDimension;++j)
233  midPlaneMatrix(i,j) = imageMidPlaneTrsf->GetMatrix()(i,j);
234  }
235 
236  HomogeneousMatrixType outputTrsfMatrix;
237  outputTrsfMatrix.SetIdentity();
238 
239  for (unsigned int i = 0;i < InputImageType::ImageDimension;++i)
240  {
241  outputTrsfMatrix(i,InputImageType::ImageDimension) = m_OutputTransform->GetOffset()[i];
242 
243  for (unsigned int j = 0;j < InputImageType::ImageDimension;++j)
244  outputTrsfMatrix(i,j) = m_OutputTransform->GetMatrix()(i,j);
245  }
246 
247  HomogeneousMatrixType matrixComposition = midPlaneMatrix * outputTrsfMatrix;
248 
249  double theta_x, theta_y, theta_z;
250  double sinus, cosinus, phi, f;
251 
252  // note : le vecteur rotation obtenu est toujours dans le plan
253  // au milieu de l'image, donc une des 3 composantes est forcement nulle
254  // ce qui donne l'idee d'une autre parametrisation du plan de symetrie
255 
256  theta_x = - matrixComposition[1][2] + matrixComposition[2][1]; //X = R(2,3) - R(3,2)
257  theta_y = - matrixComposition[2][0] + matrixComposition[0][2]; //Y = R(3,1) - R(1,3)
258  theta_z = - matrixComposition[0][1] + matrixComposition[1][0]; //Z = R(1,2) - R(2,1)
259 
260  sinus = std::sqrt(theta_x*theta_x + theta_y*theta_y + theta_z*theta_z);
261 
262  if (std::abs(sinus) > 1e-9)
263  {
264  cosinus = matrixComposition[0][0] + matrixComposition[1][1] + matrixComposition[2][2] - 1;
265  phi = std::atan(sinus / cosinus);
266  f = phi / sinus;
267  theta_x = theta_x * f;
268  theta_y = theta_y * f;
269  theta_z = theta_z * f;
270  }
271  else
272  {
273  phi = 0;
274 
275  theta_x = 0;
276  theta_y = 0;
277  theta_z = 0;
278  }
279 
280  // le vecteur rotation de r est celui de R divisé par 2
281 
282  phi = phi / 2.0;
283  theta_x = theta_x / 2.0;
284  theta_y = theta_y / 2.0;
285  theta_z = theta_z / 2.0;
286 
287  // formules : http://www.iau-sofa.rl.ac.uk/2003_0429/sofa/rv2m.for
288 
289  // calcul du vecteur unitaire ayant la même direction que le vecteur rotation
290 
291  double u_x, u_y, u_z;
292 
293  sinus = std::sqrt(theta_x*theta_x + theta_y*theta_y + theta_z*theta_z);
294 
295  if (std::abs(sinus) > 1e-9)
296  {
297  u_x = theta_x / sinus;
298  u_y = theta_y / sinus;
299  u_z = theta_z / sinus;
300  }
301 
302  else {
303  u_x = 0;
304  u_y = 0;
305  u_z = 0;
306  }
307 
308  // calcul de r
309 
310  double c, s, t;
311 
312  c = std::cos(phi);
313  s = std::sin(phi);
314  t = 1 - c;
315 
316  HomogeneousMatrixType sqrtMatrix;
317  sqrtMatrix.SetIdentity();
318 
319  sqrtMatrix[0][0] = t*u_x*u_x + c;
320  sqrtMatrix[0][1] = t*u_x*u_y - s*u_z;
321  sqrtMatrix[0][2] = t*u_x*u_z + s*u_y;
322  sqrtMatrix[1][0] = t*u_x*u_y + s*u_z;
323  sqrtMatrix[1][1] = t*u_y*u_y + c;
324  sqrtMatrix[1][2] = t*u_y*u_z - s*u_x;
325  sqrtMatrix[2][0] = t*u_x*u_z - s*u_y;
326  sqrtMatrix[2][1] = t*u_y*u_z + s*u_x;
327  sqrtMatrix[2][2] = t*u_z*u_z + c;
328 
329  // calcul de transfo5 = r+I
330 
331  MatrixType tmpMatrix;
332 
333  for (unsigned int i=0; i<3; i++)
334  for (unsigned int j=0; j<3; j++)
335  tmpMatrix[i][j] = sqrtMatrix[i][j];
336 
337  for (unsigned int i=0; i<3; i++)
338  tmpMatrix[i][i]++;
339 
340  // calcul de transfo6 = (r+I)^{-1}
341 
342  tmpMatrix = tmpMatrix.GetInverse();
343 
344  // calcul de t = (r+I)^{-1}.T
345  for (unsigned int i = 0;i < InputImageType::ImageDimension;++i)
346  for (unsigned int j = 0;j < InputImageType::ImageDimension;++j)
347  sqrtMatrix[i][3] += tmpMatrix[i][j] * matrixComposition[j][3];
348 
349  itk::Vector <double,InputImageType::ImageDimension+1> centralPointHom, centralPointTr;
350  for (unsigned int i = 0;i < InputImageType::ImageDimension;++i)
351  centralPointHom[i] = centralPoint[i];
352  centralPointHom[InputImageType::ImageDimension] = 1;
353 
354  centralPointTr = sqrtMatrix * centralPointHom;
355 
356  sqrtMatrix = sqrtMatrix.GetInverse();
357 
358  HomogeneousMatrixType trToCenterMatrix;
359  trToCenterMatrix.SetIdentity();
360 
361  for (unsigned int i = 0;i < InputImageType::ImageDimension;++i)
362  trToCenterMatrix(i,3) = centralPointTr[i] - centerReal[i];
363 
364  sqrtMatrix = sqrtMatrix * trToCenterMatrix;
365 
366  MatrixType trsfMatrix;
367  OffsetType offsetVector;
368  for (unsigned int i = 0;i < InputImageType::ImageDimension;++i)
369  {
370  offsetVector[i] = sqrtMatrix(i,3);
371  for (unsigned int j=0; j<3; j++)
372  trsfMatrix(i,j) = sqrtMatrix(i,j);
373  }
374 
375  if (m_OutputRealignTransform.IsNull())
376  m_OutputRealignTransform = BaseTransformType::New();
377 
378  m_OutputRealignTransform->SetMatrix(trsfMatrix);
379  m_OutputRealignTransform->SetOffset(offsetVector);
380 }
