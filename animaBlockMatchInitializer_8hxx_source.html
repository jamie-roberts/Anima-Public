364 template <class PixelType, unsigned int NDimensions>
365 void BlockMatchingInitializer<PixelType,NDimensions>
366 ::RegionBlockGenerator(BlockGeneratorThreadStruct *workStr, unsigned int threadId)
367 {
368  typename ImageRegionType::IndexType startPosition, blockPosition;
369  typename ImageRegionType::SizeType blockSize;
370  PointType blockOrigin;
371  ImageRegionType largestRegion = this->GetFirstReferenceImage()->GetLargestPossibleRegion();
372 
373  ImageRegionType tmpBlock;
374  int indexPos;
375  double blockVar = 0;
376  workStr->totalNumberOfBlocks[threadId] = 0;
377  workStr->blocks_variances[threadId].clear();
378  workStr->blocks_positions[threadId].clear();
379 
380  unsigned int block_half_size = std::floor ((this->GetBlockSize() - 1) / 2.0);
381 
382  bool continueLoop = true;
383  std::vector <unsigned int> positionCounter(NDimensions,0);
384 
385  while (continueLoop)
386  {
387  for (unsigned int i = 0;i < NDimensions;++i)
388  {
389  blockSize[i] = this->GetBlockSize();
390  indexPos = workStr->blockStartOffsets[i] + (workStr->startBlocks[threadId][i] + positionCounter[i])*this->GetBlockSpacing() - block_half_size;
391  if (indexPos < largestRegion.GetIndex()[i])
392  {
393  blockSize[i] += indexPos - largestRegion.GetIndex()[i];
394  indexPos = largestRegion.GetIndex()[i];
395  }
396  startPosition[i] = indexPos;
397  if (startPosition[i] + blockSize[i] > largestRegion.GetIndex()[i] + largestRegion.GetSize()[i])
398  blockSize[i] = largestRegion.GetIndex()[i] + largestRegion.GetSize()[i] - startPosition[i];
399  }
400 
401  tmpBlock.SetIndex(startPosition);
402  tmpBlock.SetSize(blockSize);
403  workStr->totalNumberOfBlocks[threadId]++;
404 
405  if (this->CheckBlockConditions(tmpBlock,blockVar,workStr,threadId))
406  {
407  for (unsigned int i = 0;i < NDimensions;++i)
408  blockPosition[i] = workStr->blockStartOffsets[i] + (workStr->startBlocks[threadId][i] + positionCounter[i])*this->GetBlockSpacing();
409 
410  if (m_GenerationMasks[workStr->maskIndex]->GetPixel(blockPosition) == 0)
411  {
412  continueLoop = this->ProgressCounter(positionCounter,workStr->nb_blocks[threadId]);
413  continue;
414  }
415 
416  workStr->tmpOutput[threadId].push_back(tmpBlock);
417  workStr->blocks_variances[threadId].push_back(blockVar);
418 
419  this->GetFirstReferenceImage()->TransformIndexToPhysicalPoint(blockPosition,blockOrigin);
420  workStr->blocks_positions[threadId].push_back(blockOrigin);
421  }
422 
423  continueLoop = this->ProgressCounter(positionCounter,workStr->nb_blocks[threadId]);
424  }
