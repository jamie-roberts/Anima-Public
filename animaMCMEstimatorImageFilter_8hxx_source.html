369 void
370 MCMEstimatorImageFilter<InputPixelType, OutputPixelType>
371 ::ComputeExtraAxonalAndKappaCoarseGrids()
372 {
373  // m_ValuesCoarseGrid[0] -> extra axonal fractions, m_ValuesCoarseGrid[1] -> kappa
374  m_ValuesCoarseGrid.resize(2);
375 
376  unsigned int coarseGridSize = 8;
377  m_ValuesCoarseGrid[0].resize(coarseGridSize);
378  m_ValuesCoarseGrid[1].resize(coarseGridSize);
379 
380  // The anisotropy index suggested by NODDI signal expression is not the ODI
381  // index defined in Zhang et al., 2012, Neuroimage. Instead, it amounts to
382  // (3 * tau1(k) - 1) / 2 \in [0,1] which is not analytically invertible.
383  // However it is well approximated by k^power / (halfLife + k^power) where
384  // power and halfLife are given by:
385 
386  double power = 1.385829;
387  double halfLife = 5.312364;
388  for (unsigned int i = 0;i < coarseGridSize;++i)
389  {
390  double tmpVal = (i + 1.0) / (coarseGridSize + 1.0);
391  m_ValuesCoarseGrid[0][i] = tmpVal;
392  m_ValuesCoarseGrid[1][i] = std::pow(tmpVal * halfLife / (1.0 - tmpVal), 1.0 / power);
393  }

1044  MCMType::ListType &workVec,ParametersType &p)

1047  unsigned int numIsoCompartments = mcmUpdateValue->GetNumberOfIsotropicCompartments();
1048  unsigned int numCompartments = mcmUpdateValue->GetNumberOfCompartments();
1049 
1050  double optRadialDiff1 = 0.0;
1051  double optRadialDiff2 = 0.0;
1052  double optAzimuth = 0.0;
1053  double optValue = - 1.0;
1054 
1055  double meanAxialDiff = 0.0;
1056  double meanRadialDiff = 0.0;
1057  unsigned int numUsefulCompartments = 0;
1058 
1059  for (unsigned int i = numIsoCompartments;i < numCompartments;++i)
1060  {
1061  if (mcmUpdateValue->GetCompartmentWeight(i) > 0)
1062  {
1063  meanAxialDiff += mcmUpdateValue->GetCompartment(i)->GetAxialDiffusivity();
1064  meanRadialDiff += mcmUpdateValue->GetCompartment(i)->GetRadialDiffusivity1();
1065 
1066  ++numUsefulCompartments;
1067  }
1068  }
1069 
1070  if (numUsefulCompartments > 0)
1071  {
1072  meanAxialDiff /= numUsefulCompartments;
1073  meanRadialDiff /= numUsefulCompartments;
1074  }
1075  else
1076  {
1077  meanAxialDiff = m_AxialDiffusivityValue;
1078  meanRadialDiff = (m_RadialDiffusivity1Value + m_RadialDiffusivity2Value) / 2.0;
1079  }
1080 
1081  if (meanAxialDiff - meanRadialDiff < anima::MCMAxialDiffusivityAddonLowerBound)
1082  meanAxialDiff = meanRadialDiff + anima::MCMAxialDiffusivityAddonLowerBound;
1083 
1084  for (unsigned int i = numIsoCompartments;i < numCompartments;++i)
1085  mcmUpdateValue->GetCompartment(i)->SetAxialDiffusivity(meanAxialDiff);
1086

1088  {

1090

1092  double tmpRadialDiffusivity2 = 0.5 * ((1.0 + tmpWeight2) * meanRadialDiff + (1.0 - tmpWeight2) * anima::MCMDiffusivityLowerBound);
1093 
1094  for (unsigned int k = 0;k < m_ValuesCoarseGrid[1].size();++k)
1095  {
1096  double tmpWeight1 = m_ValuesCoarseGrid[1][k];
1097 
1098  // In between meanRD and (meanRD + meanAD) / 2
1099  double tmpRadialDiffusivity1 = 0.5 * ((1.0 + tmpWeight1) * meanRadialDiff + (1.0 - tmpWeight1) * meanAxialDiff);
1100  if (meanAxialDiff - tmpRadialDiffusivity1 < anima::MCMAxialDiffusivityAddonLowerBound)
1101  continue;
1102 
1103  for (unsigned int j = 0;j < m_ValuesCoarseGrid[0].size();++j)
1104  {
1105  double tmpAzimuth = m_ValuesCoarseGrid[0][j];
1106 
1107  for (unsigned int i = numIsoCompartments;i < numCompartments;++i)
1108  {
1109  mcmUpdateValue->GetCompartment(i)->SetPerpendicularAngle(tmpAzimuth);
1110  mcmUpdateValue->GetCompartment(i)->SetRadialDiffusivity1(tmpRadialDiffusivity1);
1111  mcmUpdateValue->GetCompartment(i)->SetRadialDiffusivity2(tmpRadialDiffusivity2);
1112  }
1113 
1114  workVec = mcmUpdateValue->GetParametersAsVector();
1115  for (unsigned int i = 0;i < dimension;++i)
1116  p[i] = workVec[i];
1117 
1118  double tmpValue = this->GetCostValue(cost,p);
1119 
1120  if ((tmpValue < optValue) || (optValue < 0))
1121  {
1122  optValue = tmpValue;
1123  optRadialDiff1 = tmpRadialDiffusivity1;
1124  optRadialDiff2 = tmpRadialDiffusivity2;
1125  optAzimuth = tmpAzimuth;
1126  }
1127  }
1128  }
1129  }
1130 
1131  for (unsigned int i = numIsoCompartments;i < numCompartments;++i)
1132  {
