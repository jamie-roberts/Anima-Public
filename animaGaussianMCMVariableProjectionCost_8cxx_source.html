273 {
274  unsigned int nbParams = parameters.GetSize();
275  if (nbParams == 0)
276  return;
277 
278  unsigned int nbValues = m_ObservedSignals.size();
279  unsigned int numCompartments = m_IndexesUsefulCompartments.size();
280 
281  this->PrepareDataForDerivative();
282  unsigned int numOnCompartments = m_GramMatrix.rows();
283 
284  // Assume get derivative is called with the same parameters as GetValue just before
285  for (unsigned int i = 0;i < nbParams;++i)
286  {
287  if (m_TestedParameters[i] != parameters[i])
288  {
289  std::cerr << parameters << std::endl;
290  itkExceptionMacro("Get derivative not called with the same parameters as GetValue, suggestive of NaN...");
291  }
292  }
293 
294  derivative.SetSize(nbParams, nbValues);
295  derivative.Fill(0.0);
296  ListType DFw(nbValues,0.0);
297 
298  ListType tmpVec(numOnCompartments,0.0);
299 
300  for (unsigned int k = 0;k < nbParams;++k)
301  {
302  // First, compute
303  // - DFw = DF[k] w
304  std::fill(DFw.begin(),DFw.end(),0.0);
305  for (unsigned int i = 0;i < nbValues;++i)
306  {
307  DFw[i] = 0.0;
308  for (unsigned int j = 0;j < numCompartments;++j)
309  DFw[i] += m_SignalAttenuationsJacobian[k](i,j) * m_OptimalUsefulWeights[j];
310  }
311 
312  // Then, compute
313  // - tmpVec = F^T DF[k] w - (DF[k])^T Residuals
314  unsigned int pos = 0;
315  for (unsigned int j = 0;j < numCompartments;++j)
316  {
317  if (!m_CompartmentSwitches[j])
318  continue;
319 
320  tmpVec[pos] = 0.0;
321 
322  for (unsigned int i = 0;i < nbValues;++i)
323  tmpVec[pos] += m_PredictedSignalAttenuations(i,j) * DFw[i] - m_SignalAttenuationsJacobian[k](i,j) * m_Residuals[i];
324 
325  ++pos;
326  }
327 
328  // Finally, get
329  // - derivative = FMatrixInverseG tmpVec - DFw
330  for (unsigned int i = 0;i < nbValues;++i)
331  {
332  derivative(k,i) = - DFw[i];
333 
334  for (unsigned int j = 0;j < numOnCompartments;++j)
335  derivative(k,i) += m_FMatrixInverseG(i,j) * tmpVec[j];
336  }
337  }
338

341  {
342  if (!boost::math::isfinite(derivative(i,0)))
343  {
344  problem = true;
345  break;
346  }
347  }
348 
349  if (problem)

351  std::cerr << "Derivative: " << derivative << std::endl;
352  std::cerr << "Optimal weights: " << m_OptimalUsefulWeights << std::endl;
