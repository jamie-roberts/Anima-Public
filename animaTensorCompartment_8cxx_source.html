29 {
30  this->UpdateDiffusionTensor();
31 
32  m_JacobianVector.resize(this->GetNumberOfParameters());
33 
34  double signalAttenuation = this->GetFourierTransformedDiffusionProfile(smallDelta, bigDelta, gradientStrength, gradient);
35  double innerProd1 = anima::ComputeScalarProduct(gradient, m_EigenVector1);
36  double innerProd2 = anima::ComputeScalarProduct(gradient, m_EigenVector2);
37 
38  double DgTe1DTheta = m_CosTheta * (gradient[0] * m_CosPhi + gradient[1] * m_SinPhi) - gradient[2] * m_SinTheta;
39  double DgTe1DPhi = m_SinTheta * (gradient[1] * m_CosPhi - gradient[0] * m_SinPhi);
40 
41  double DgTe2DTheta = m_SinAlpha * innerProd1;
42  double DgTe2DPhi = gradient[0] * (m_CosTheta * m_SinPhi * m_SinAlpha - m_CosPhi * m_CosAlpha) - gradient[1] * (m_SinPhi * m_CosAlpha + m_CosTheta * m_CosPhi * m_SinAlpha);
43  double DgTe2DAlpha = gradient[0] * (m_SinPhi * m_SinAlpha - m_CosTheta * m_CosPhi * m_CosAlpha) - gradient[1] * (m_CosPhi * m_SinAlpha + m_CosTheta * m_SinPhi * m_CosAlpha) + gradient[2] * m_SinTheta * m_CosAlpha;
44 
45  double diffAxialRadial2 = this->GetAxialDiffusivity() - this->GetRadialDiffusivity2();
46  double diffRadialDiffusivities = this->GetRadialDiffusivity1() - this->GetRadialDiffusivity2();
47 
48  // Derivative w.r.t. theta
49  double bValue = anima::GetBValueFromAcquisitionParameters(smallDelta, bigDelta, gradientStrength);
50  m_JacobianVector[0] = -2.0 * bValue * (diffAxialRadial2 * innerProd1 * DgTe1DTheta + diffRadialDiffusivities * innerProd2 * DgTe2DTheta) * signalAttenuation;
51 
52  // Derivative w.r.t. phi
53  m_JacobianVector[1] = -2.0 * bValue * (diffAxialRadial2 * innerProd1 * DgTe1DPhi + diffRadialDiffusivities * innerProd2 * DgTe2DPhi) * signalAttenuation;
54 
55  // Derivative w.r.t. alpha
56  m_JacobianVector[2] = -2.0 * bValue * diffRadialDiffusivities * innerProd2 * DgTe2DAlpha * signalAttenuation;
57 
58  if (m_EstimateDiffusivities)
59  {
60  // Derivative w.r.t. to d1
61  m_JacobianVector[3] = - bValue * innerProd1 * innerProd1 * signalAttenuation;
62 
63  // Derivative w.r.t. to d2
64  m_JacobianVector[4] = - bValue * (innerProd1 * innerProd1 + innerProd2 * innerProd2) * signalAttenuation;
65 
66  // Derivative w.r.t. to d3
67  m_JacobianVector[5] = - bValue * signalAttenuation;
68  }

281 void TensorCompartment::UpdateParametersFromTensor()
282 {
283  if (!m_UpdatedCompartment)
284  return;
285 
286  vnl_matrix <double> eVecs(m_SpaceDimension,m_SpaceDimension,0);
287  vnl_diag_matrix <double> eVals(m_SpaceDimension);
288 
289  itk::SymmetricEigenAnalysis < Matrix3DType,vnl_diag_matrix <double>,vnl_matrix <double> > eigSys(m_SpaceDimension);
290 
291  eigSys.ComputeEigenValuesAndVectors(m_DiffusionTensor,eVals,eVecs);
292 
293  Vector3DType e1, sphDir;
294 
295  for (unsigned int i = 0;i < m_SpaceDimension;++i)
296  e1[i] = eVecs(2,i);
297 
298  anima::TransformCartesianToSphericalCoordinates(e1,sphDir);
299 
300  double theta = sphDir[0];
301  double phi = sphDir[1];
302 
303  double alpha = 0;
304  if ((theta != 0)&&(theta != M_PI))
305  alpha = std::atan2(eVecs(1,2),- eVecs(0,2));
306  else
307  {
308  // Use two first components of 2nd eigenvector
309  // Infinite number of solutions for phi from main eigenvector, set it as alpha - pi/2
310  alpha = std::atan2(1 - eVecs(1,0), eVecs(1,1));
311  phi = alpha - M_PI / 2.0;
312  }
313 
314  if (alpha < 0)
315  alpha += 2.0 * M_PI;
316 
317  if (phi < 0)
318  phi += 2.0 * M_PI;
319 
320  m_UpdatedCompartment = false;
321 
322  this->SetOrientationTheta(theta);
323  this->SetOrientationPhi(phi);
324  this->SetPerpendicularAngle(alpha);
325 
326  double axialDiff = std::max(anima::MCMDiffusivityLowerBound,std::min(eVals[2],anima::MCMDiffusivityUpperBound));
327  double radialDiff1 = std::max(anima::MCMDiffusivityLowerBound,std::min(eVals[1],anima::MCMDiffusivityUpperBound));
328  double radialDiff2 = std::max(anima::MCMDiffusivityLowerBound,std::min(eVals[0],anima::MCMDiffusivityUpperBound));
329 
330  this->SetAxialDiffusivity(axialDiff);
331  this->SetRadialDiffusivity1(radialDiff1);
332  this->SetRadialDiffusivity2(radialDiff2);

400 void TensorCompartment::UpdateDiffusionTensor()
401 {
402  if (!m_ModifiedTensor)
403  return;
404 
405  this->UpdateAngleConfiguration();
406  double axialDiff = this->GetAxialDiffusivity();
407  double radialDiff1 = this->GetRadialDiffusivity1();
408  double radialDiff2 = this->GetRadialDiffusivity2();
409 
410  m_DiffusionTensor.Fill(0.0);
411 
412  for (unsigned int i = 0;i < m_SpaceDimension;++i)
413  m_DiffusionTensor(i,i) = radialDiff2;
414 
415  for (unsigned int i = 0;i < m_SpaceDimension;++i)
416  for (unsigned int j = i;j < m_SpaceDimension;++j)
417  {
418  m_DiffusionTensor(i,j) += m_EigenVector1[i] * m_EigenVector1[j] * (axialDiff - radialDiff2);
419  m_DiffusionTensor(i,j) += m_EigenVector2[i] * m_EigenVector2[j] * (radialDiff1 - radialDiff2);
420 
421  if (i != j)
422  m_DiffusionTensor(j,i) = m_DiffusionTensor(i,j);
423  }
424 
425  m_TensorDeterminant = axialDiff * radialDiff1 * radialDiff2;
426 
427  m_UpdateInverseTensor = true;
428  m_ModifiedTensor = false;
429 }

476 double TensorCompartment::GetFractionalAnisotropy()
477 {

479  double l2 = this->GetRadialDiffusivity1();
480  double l3 = this->GetRadialDiffusivity2();
481

483  double denomFA = std::sqrt (l1 * l1 + l2 * l2 + l3 * l3);
