338 double
339 DTIEstimationImageFilter<InputPixelScalarType, OutputPixelScalarType>
340 ::ComputeB0AndVarianceFromTensorVector(const vnl_matrix <double> &tensorValue, const std::vector <double> &dwiSignal, double &outVarianceValue)
341 {
342  double sumSquaredObservedSignals = 0;
343  double sumPredictedPerObservedSignals = 0;
344  double sumSquaredPredictedSignals = 0;
345  unsigned int numInputs = dwiSignal.size();
346 
347  for (unsigned int i = 0;i < numInputs;++i)
348  {
349  double observedValue = dwiSignal[i];
350  double predictedValue = 0;
351  double bValue = m_BValuesList[i];
352 
353  if (bValue == 0)
354  predictedValue = 1;
355  else
356  {
357  for (unsigned int j = 0;j < 3;++j)
358  {
359  predictedValue += tensorValue(j,j) * m_GradientDirections[i][j] * m_GradientDirections[i][j];
360  for (unsigned int k = j + 1;k < 3;++k)
361  predictedValue += 2.0 * tensorValue(j,k) * m_GradientDirections[i][j] * m_GradientDirections[i][k];
362  }
363 
364  predictedValue = std::exp(- bValue * predictedValue);
365  }
366 
367  sumSquaredObservedSignals += observedValue * observedValue;
368  sumSquaredPredictedSignals += predictedValue * predictedValue;
369  sumPredictedPerObservedSignals += predictedValue * observedValue;
370  }
371 
372  double outB0Value = sumPredictedPerObservedSignals / sumSquaredPredictedSignals;
373  outVarianceValue = (sumSquaredObservedSignals - 2.0 * outB0Value * sumPredictedPerObservedSignals + outB0Value * outB0Value * sumSquaredPredictedSignals) / (numInputs - 1.0);
374  return outB0Value;
375 }
